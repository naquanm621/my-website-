function selfDestruct() {

    const elements = Array.from(document.querySelectorAll('body *'));

    document.getElementById('mood').innerText = "PANIC";

   

    // Reverse the array so it deletes from the bottom up (less glitchy)

    elements.reverse();

    let i = 0;

    const interval = setInterval(() => {

        if (i < elements.length) {

            elements[i].style.transition = "all 0.5s ease";

            elements[i].style.opacity = "0";

            elements[i].style.transform = "scale(0)";

           

            // Remove the element after the animation

            setTimeout(() => elements[i-1]?.remove(), 500);

            i++;

        } else {

            clearInterval(interval);

            document.body.innerHTML = "<h1 style='color: white; margin-top: 20%;'>I'm gone. I hope you're happy.</h1>";

            document.body.style.backgroundColor = "black";

        }

    }, 100); // Deletes an element every 100ms

}
